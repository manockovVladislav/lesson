k,n = [int(i) for i in input().split()]
a = [i for i in range(k)]
print(*a)

def one(a, k, n):
    for i in range(k-1, -1, -1):
        if a[i] < (n-1) - (k-1-i):
            a[i] += 1
            for j in range(i+1,k):
                a[j] = a[j - 1] + 1
            print(*a)
            one(a, k, n)
one(a, k, n)
